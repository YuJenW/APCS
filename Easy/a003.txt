#include <stdio.h>

int main()
{
    int M, D;
    scanf("%d %d",&M,&D);
    if ((M*2+D)%3==0)
            printf("普通");
    else if ((M*2+D)%3==1)
            printf("吉");
    else if ((M*2+D)%3==2)
            printf("大吉");

    return 0;
}
